## Threats
- **T1: GP fits transit signal as noise** (severity: medium) — Without transit masking, GP will model the transit dip. Mitigation: Always mask in-transit points (from TLS ephemeris) before GP fitting; predict on all time points.
- **T2: GP hyperparameter optimization diverges** (severity: low) — Nelder-Mead may not converge for some stars. Mitigation: Bound sigma to [1e-5, 0.1] and rho to [0.1, 10.0]; fall back to biweight result if optimization fails.
- **T3: ExoFOP label download unavailable** (severity: low) — ExoFOP API may change or be down. Mitigation: Cache downloaded labels as CSV in data/reference/; retry with backoff.
</threat_model>

```python
"""GP detrending module — celerite2 Matérn-3/2 for top SDE≥7 candidates (PREP-04)."""
import numpy as np
import pandas as pd
import logging
from pathlib import Path
from scipy.optimize import minimize
from tqdm import tqdm
from celerite2 import GaussianProcess
from celerite2.terms import Matern32Term
from transitleastsquares import transit_mask


def gp_detrend_single(time: np.ndarray, flux: np.ndarray, flux_err: np.ndarray,
                      period: float, duration: float, t0: float,
                      sigma_init: float = 0.001, rho_init: float = 0.5
                      ) -> tuple:
    """Apply GP detrending with transit masking on a single light curve.
    
    Returns (gp_detrended_flux, gp_trend, sigma_opt, rho_opt) or
    (None, None, None, None) on failure.
    """
    # Mask in-transit points before GP fitting
    in_transit = transit_mask(time, period, duration, t0)
    time_oot = time[~in_transit]
    flux_oot = flux[~in_transit]
    flux_err_oot = flux_err[~in_transit]
    
    if len(time_oot) < 50:
        return None, None, None, None
    
    # Sort by time (celerite2 requirement)
    sort_idx = np.argsort(time_oot)
    time_oot = time_oot[sort_idx]
    flux_oot = flux_oot[sort_idx]
    flux_err_oot = flux_err_oot[sort_idx]
    
    # Build GP with Matérn-3/2 kernel
    kernel = Matern32Term(sigma=sigma_init, rho=rho_init)
    gp = GaussianProcess(kernel, mean=1.0)
    gp.compute(time_oot, yerr=flux_err_oot)
    
    # Optimize hyperparameters
    def neg_log_likelihood(params):
        sigma, rho = np.exp(params)
        # Bound checks
        if sigma < 1e-5 or sigma > 0.1 or rho < 0.1 or rho > 10.0:
            return 1e10
        gp.kernel = Matern32Term(sigma=sigma, rho=rho)
        try:
            gp.recompute()
            return -gp.log_likelihood(flux_oot)
        except Exception:
            return 1e10
    
    x0 = np.log([sigma_init, rho_init])
    result = minimize(neg_log_likelihood, x0, method="Nelder-Mead",
                      options={"maxiter": 500})
    
    if not result.success:
        return None, None, None, None
    
    sigma_opt, rho_opt = np.exp(result.x)
    
    # Predict GP trend on ALL time points (including in-transit)
    sort_all = np.argsort(time)
    time_sorted = time[sort_all]
    
    gp.kernel = Matern32Term(sigma=sigma_opt, rho=rho_opt)
    gp.compute(time_oot, yerr=flux_err_oot)
    mu_all = gp.predict(flux_oot, t=time_sorted, return_var=False)
    
    # Unsort to original order
    unsort_idx = np.argsort(sort_all)
    gp_trend = mu_all[unsort_idx]
    
    # Divide out GP trend
    gp_detrended = flux / gp_trend
    
    return gp_detrended, gp_trend, sigma_opt, rho_opt


def run_gp_detrend(config: dict, logger: logging.Logger) -> None:
    """Run GP detrending on top N SDE≥7 candidates."""
    paths = config["paths"]
    gp_cfg = config["gp_detrending"]
    top_n = gp_cfg["top_n_candidates"]
    
    # Load TLS results
    tls_path = Path(paths["catalogue"]) / "tls_results.parquet"
    if not tls_path.exists():
        logger.warning("No TLS results found — skipping GP detrending")
        return
    
    tls_df = pd.read_parquet(str(tls_path))
    full_candidates = tls_df[tls_df["sde_tier"] == "full"].copy()
    full_candidates = full_candidates.sort_values("sde", ascending=False).head(top_n)
    
    logger.info(f"GP detrending top {len(full_candidates)} candidates")
    
    success, failed = 0, 0
    for _, row in tqdm(full_candidates.iterrows(), total=len(full_candidates),
                       desc="GP detrend"):
        tic_id = int(row["tic_id"])
        sector = int(row["sector"])
        
        input_path = (Path(paths["preprocessed"]) / f"sector{sector}" /
                      f"TIC_{tic_id}_preprocessed.npz")
        output_path = (Path(paths["preprocessed"]) / f"sector{sector}" /
                       f"TIC_{tic_id}_gp_detrended.npz")
        
        if output_path.exists():
            success += 1
            continue
        
        if not input_path.exists():
            failed += 1
            continue
        
        try:
            data = np.load(str(input_path))
            time = data["time"]
            flux = data["flux_normalized"]  # Use pre-detrending normalized flux
            flux_err = data["flux_err"]
            
            gp_flux, gp_trend, sigma, rho = gp_detrend_single(
                time, flux, flux_err,
                period=float(row["period"]),
                duration=float(row["duration"]),
                t0=float(row["t0"]),
                sigma_init=gp_cfg["sigma_init"],
                rho_init=gp_cfg["rho_init"],
            )
            
            if gp_flux is not None:
                np.savez_compressed(
                    str(output_path),
                    time=time.astype(np.float64),
                    flux=gp_flux.astype(np.float32),
                    flux_err=flux_err.astype(np.float32),
                    gp_trend=gp_trend.astype(np.float32),
                    gp_sigma=np.float32(sigma),
                    gp_rho=np.float32(rho),
                )
                success += 1
                logger.info(
                    f"GP detrended TIC {tic_id} (sigma={sigma:.4f}, rho={rho:.3f})",
                    extra={"tic_id": tic_id, "sector": sector, "status": "success"}
                )
            else:
                failed += 1
                logger.warning(f"GP optimization failed for TIC {tic_id}")
        
        except Exception as e:
            failed += 1
            logger.warning(f"GP detrend failed TIC {tic_id}: {e}")
    
    logger.info(f"GP detrending complete: {success} success, {failed} failed")
```

```python
"""Validation target recovery tests for Phase 1 pipeline.

These tests verify that the pipeline correctly detects known exoplanets
in their respective TESS sectors. Run after full pipeline execution.

Usage: python -m pytest tests/test_validation_targets.py -v
"""
import numpy as np
import pandas as pd
import pytest
from pathlib import Path


# Known validation targets
WASP_121B = {
    "tic_id": 22529346,
    "sector": 1,
    "period": 1.2749255,
    "period_tolerance": 0.0001,  # 0.01%
    "min_sde": 30.0,
    "description": "Ultra-hot Jupiter, deep transit",
}

TOI_270 = {
    "tic_id": 259377017,
    "sector": 3,
    "planets": [
        {"name": "b", "period": 3.360, "min_sde": 5.0, "period_tol": 0.01},
        {"name": "c", "period": 5.660, "min_sde": 7.0, "period_tol": 0.01},
        {"name": "d", "period": 11.380, "min_sde": 5.0, "period_tol": 0.05},
    ],
    "description": "3-planet system, tests multi-planet iterative search",
}

L_98_59 = {
    "tic_id": 307210830,
    "sector": 2,
    "planets": [
        {"name": "b", "period": 2.2531, "min_sde": 5.0, "period_tol": 0.01},
        {"name": "c", "period": 3.6904, "min_sde": 7.0, "period_tol": 0.01},
        {"name": "d", "period": 7.4512, "min_sde": 5.0, "period_tol": 0.02},
    ],
    "description": "Earth-sized planet test, shallow 400 ppm transit",
}


TLS_RESULTS_PATH = "data/catalogue/tls_results.parquet"


@pytest.fixture
def tls_results():
    """Load TLS results Parquet."""
    path = Path(TLS_RESULTS_PATH)
    if not path.exists():
        pytest.skip("TLS results not yet generated")
    return pd.read_parquet(path)


class TestWASP121b:
    """VAL-01: WASP-121b detection in Sector 1."""

    def test_detected_with_high_sde(self, tls_results):
        """WASP-121b must be detected with SDE > 30."""
        matches = tls_results[
            (tls_results["tic_id"] == WASP_121B["tic_id"]) &
            (tls_results["sector"] == WASP_121B["sector"])
        ]
        assert len(matches) > 0, "WASP-121b not found in TLS results"
        best = matches.loc[matches["sde"].idxmax()]
        assert best["sde"] >= WASP_121B["min_sde"], (
            f"WASP-121b SDE={best['sde']:.1f}, expected >= {WASP_121B['min_sde']}"
        )

    def test_period_accuracy(self, tls_results):
        """WASP-121b period must be within 0.01% of published value."""
        matches = tls_results[
            (tls_results["tic_id"] == WASP_121B["tic_id"]) &
            (tls_results["sector"] == WASP_121B["sector"])
        ]
        best = matches.loc[matches["sde"].idxmax()]
        period_err = abs(best["period"] - WASP_121B["period"]) / WASP_121B["period"]
        assert period_err < WASP_121B["period_tolerance"], (
            f"WASP-121b period error={period_err:.5f}, expected < {WASP_121B['period_tolerance']}"
        )

    def test_sde_tier_full(self, tls_results):
        """WASP-121b must be classified as 'full' SDE tier."""
        matches = tls_results[
            (tls_results["tic_id"] == WASP_121B["tic_id"]) &
            (tls_results["sector"] == WASP_121B["sector"])
        ]
        best = matches.loc[matches["sde"].idxmax()]
        assert best["sde_tier"] == "full"


class TestTOI270:
    """VAL-02: TOI-270 multi-planet recovery in Sector 3."""

    def test_all_three_planets_detected(self, tls_results):
        """All 3 TOI-270 planets must be detected with SDE >= 5."""
        matches = tls_results[
            (tls_results["tic_id"] == TOI_270["tic_id"]) &
            (tls_results["sector"] == TOI_270["sector"])
        ]
        assert len(matches) >= 3, (
            f"TOI-270: found {len(matches)} signals, expected >= 3"
        )
        for planet in TOI_270["planets"]:
            period_matches = matches[
                (abs(matches["period"] - planet["period"]) / planet["period"] < planet["period_tol"]) &
                (matches["sde"] >= planet["min_sde"])
            ]
            assert len(period_matches) > 0, (
                f"TOI-270 {planet['name']} (P={planet['period']}d) not recovered"
            )


class TestL9859:
    """VAL-02: L 98-59 multi-planet + shallow transit recovery in Sector 2."""

    def test_planets_c_and_d_detected(self, tls_results):
        """L 98-59 c and d must be detected with SDE >= 7."""
        matches = tls_results[
            (tls_results["tic_id"] == L_98_59["tic_id"]) &
            (tls_results["sector"] == L_98_59["sector"])
        ]
        for planet in L_98_59["planets"][1:]:  # c and d
            period_matches = matches[
                (abs(matches["period"] - planet["period"]) / planet["period"] < planet["period_tol"]) &
                (matches["sde"] >= planet["min_sde"])
            ]
            assert len(period_matches) > 0, (
                f"L 98-59 {planet['name']} (P={planet['period']}d) not recovered"
            )

    def test_planet_b_at_least_subthreshold(self, tls_results):
        """L 98-59 b (400 ppm) must be detected with SDE >= 5."""
        matches = tls_results[
            (tls_results["tic_id"] == L_98_59["tic_id"]) &
            (tls_results["sector"] == L_98_59["sector"])
        ]
        planet = L_98_59["planets"][0]  # planet b
        period_matches = matches[
            (abs(matches["period"] - planet["period"]) / planet["period"] < planet["period_tol"]) &
            (matches["sde"] >= planet["min_sde"])
        ]
        assert len(period_matches) > 0, (
            f"L 98-59 b (P={planet['period']}d, 400 ppm) not recovered at SDE >= 5"
        )


class TestDetrendingPreservation:
    """Verify detrending does not erase transit signals."""

    def test_wasp121b_sde_preserved(self, tls_results):
        """WASP-121b post-detrend SDE must remain strong (>= 30)."""
        matches = tls_results[
            (tls_results["tic_id"] == WASP_121B["tic_id"]) &
            (tls_results["sector"] == WASP_121B["sector"])
        ]
        if len(matches) == 0:
            pytest.fail("WASP-121b not in results — cannot check preservation")
        best = matches.loc[matches["sde"].idxmax()]
        assert best["sde"] >= 30.0, (
            f"WASP-121b SDE after detrending = {best['sde']:.1f}, expected >= 30"
        )
```

```python
"""Download ExoFOP-TESS TOI disposition table for training labels (DATA-03).

Usage: python scripts/download_exofop.py
Output: data/reference/exofop_toi_dispositions.csv
"""
import pandas as pd
from pathlib import Path


EXOFOP_TOI_URL = "https://exofop.ipac.caltech.edu/tess/download_toi.php?sort=toi&output=csv"


def main():
    output_path = Path("data/reference/exofop_toi_dispositions.csv")
    output_path.parent.mkdir(parents=True, exist_ok=True)

    print("Downloading ExoFOP-TESS TOI disposition table...")
    try:
        df = pd.read_csv(EXOFOP_TOI_URL)
        df.to_csv(str(output_path), index=False)
        print(f"Saved {len(df)} TOIs to {output_path}")
        print(f"Columns: {list(df.columns[:10])}...")
        
        # Summary of dispositions
        if "TFOPWG Disposition" in df.columns:
            print("\nDisposition distribution:")
            print(df["TFOPWG Disposition"].value_counts())
    except Exception as e:
        print(f"Download failed: {e}")
        print("Fallback: manually download from ExoFOP and place at:")
        print(f"  {output_path}")


if __name__ == "__main__":
    main()
```
