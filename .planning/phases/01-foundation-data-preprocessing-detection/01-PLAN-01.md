"""Configuration loader for the exoplanet detection pipeline."""
import yaml
from pathlib import Path
from dataclasses import dataclass, field
from typing import List


def load_config(config_path: str = "config.yaml") -> dict:
    """Load pipeline configuration from YAML file."""
    path = Path(config_path)
    if not path.exists():
        raise FileNotFoundError(f"Config file not found: {config_path}")
    with open(path, "r") as f:
        config = yaml.safe_load(f)
    return config


def ensure_directories(config: dict) -> None:
    """Create all pipeline directories from config paths section."""
    paths = config["paths"]
    sectors = config["pipeline"]["sectors"]
    for key, base_path in paths.items():
        Path(base_path).mkdir(parents=True, exist_ok=True)
    # Create per-sector subdirectories for raw, preprocessed, tls
    for sector in sectors:
        for stage in ["raw", "preprocessed", "tls"]:
            Path(paths[stage]) / f"sector{sector}").mkdir(parents=True, exist_ok=True)
    # Ensure catalogue and logs directories exist
    Path(paths["catalogue"]).mkdir(parents=True, exist_ok=True)
    Path(paths["logs"]).mkdir(parents=True, exist_ok=True)
    Path(paths["reference"]).mkdir(parents=True, exist_ok=True)
    Path(paths["outputs"]).mkdir(parents=True, exist_ok=True)
```

"""Structured JSON-lines logging for the exoplanet detection pipeline."""
import json
import logging
import time
from pathlib import Path
from datetime import datetime, timezone


class JsonLinesHandler(logging.Handler):
    """Logging handler that writes JSON-lines format to a file."""

    def __init__(self, log_path: str):
        super().__init__()
        Path(log_path).parent.mkdir(parents=True, exist_ok=True)
        self.file = open(log_path, "a")

    def emit(self, record: logging.LogRecord):
        entry = {
            "timestamp": datetime.now(timezone.utc).isoformat(),
            "level": record.levelname,
            "module": record.module,
            "message": record.getMessage(),
        }
        if hasattr(record, "tic_id"):
            entry["tic_id"] = record.tic_id
        if hasattr(record, "sector"):
            entry["sector"] = record.sector
        if hasattr(record, "duration_s"):
            entry["duration_s"] = record.duration_s
        if hasattr(record, "status"):
            entry["status"] = record.status
        self.file.write(json.dumps(entry) + "\n")
        self.file.flush()

    def close(self):
        self.file.close()
        super().close()


def setup_logging(log_dir: str = "data/logs") -> logging.Logger:
    """Configure pipeline logger with JSON-lines file + console output."""
    logger = logging.getLogger("pipeline")
    logger.setLevel(logging.INFO)
    # JSON-lines file handler
    log_path = Path(log_dir) / "pipeline.log"
    jl_handler = JsonLinesHandler(str(log_path))
    logger.addHandler(jl_handler)
    # Console handler (human-readable)
    console = logging.StreamHandler()
    console.setFormatter(logging.Formatter("[%(levelname)s] %(message)s"))
    logger.addHandler(console)
    return logger


class Timer:
    """Context manager for timing pipeline stages."""

    def __init__(self):
        self.elapsed = 0.0

    def __enter__(self):
        self._start = time.perf_counter()
        return self

    def __exit__(self, *args):
        self.elapsed = time.perf_counter() - self._start
```
