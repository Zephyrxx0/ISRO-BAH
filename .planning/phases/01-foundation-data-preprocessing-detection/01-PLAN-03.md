## Threats
- **T1: Biweight detrending erases shallow transits** (severity: medium) — Window too short could attenuate 50-200 ppm signals. Mitigation: Window_length=0.75d is ≥6× typical hot Jupiter T14 (2.88h); validated against WASP-121b and L 98-59 b in Plan 05.
- **T2: Sigma-clipping removes legitimate transit dips** (severity: medium) — Aggressive clipping could flag deep transits as outliers. Mitigation: 5σ threshold is safe for transits (typical transit = 1-3σ from median); only extreme outliers removed.
- **T3: Corrupted .npz files crash pipeline** (severity: low) — Malformed input data. Mitigation: Wrap per-star processing in try/except; log and skip failures; report summary at end.
</threat_model>

```python
"""Preprocessing module — quality masking, sigma-clipping, normalization, detrending."""
import numpy as np
import logging
from pathlib import Path
from multiprocessing import Pool
from functools import partial
from tqdm import tqdm
from wotan import flatten

from src.logging_utils import Timer


# TESS quality bitmask (PREP-01)
# AttitudeTweak|SafeMode|CoarsePoint|EarthPoint|Argabrightening|ApertureCosmic|ManualExclude|SensitivityDropout|ImpulsiveOutlier
QUALITY_BITMASK = 1 | 2 | 4 | 8 | 16 | 64 | 128 | 256 | 512  # = 991


def apply_quality_mask(time: np.ndarray, flux: np.ndarray,
                       flux_err: np.ndarray, quality: np.ndarray
                       ) -> tuple:
    """Remove cadences with quality flags set (PREP-01)."""
    mask = (quality & QUALITY_BITMASK) == 0
    return time[mask], flux[mask], flux_err[mask]


def sigma_clip(flux: np.ndarray, flux_err: np.ndarray, time: np.ndarray,
               sigma: float = 5.0) -> tuple:
    """Remove outliers beyond sigma threshold from median (PREP-02)."""
    median = np.nanmedian(flux)
    std = np.nanstd(flux)
    mask = np.abs(flux - median) < sigma * std
    return time[mask], flux[mask], flux_err[mask]


def normalize_flux(flux: np.ndarray, flux_err: np.ndarray) -> tuple:
    """Normalize flux to median = 1.0 (PREP-02)."""
    median = np.nanmedian(flux)
    return flux / median, flux_err / median


def mask_gap_edges(time: np.ndarray, flux: np.ndarray, flux_err: np.ndarray,
                   gap_threshold: float = 0.5, edge_cadences: int = 2
                   ) -> tuple:
    """Flag cadences adjacent to data gaps as unreliable (PREP-05).
    
    Returns arrays with gap-edge cadences removed.
    """
    dt = np.diff(time)
    gap_indices = np.where(dt > gap_threshold)[0]
    
    if len(gap_indices) == 0:
        return time, flux, flux_err
    
    mask = np.ones(len(time), dtype=bool)
    for idx in gap_indices:
        # Mark edge_cadences before and after each gap
        start = max(0, idx - edge_cadences + 1)
        end = min(len(time), idx + edge_cadences + 1)
        mask[start:idx + 1] = False
        mask[idx + 1:end] = False
    
    return time[mask], flux[mask], flux_err[mask]


def biweight_detrend(time: np.ndarray, flux: np.ndarray,
                     window_length: float = 0.75,
                     break_tolerance: float = 0.5,
                     cval: float = 5.0) -> tuple:
    """Detrend with Tukey's biweight (Wotan) preserving transits (PREP-03).
    
    Returns (detrended_flux, trend).
    """
    detrended, trend = flatten(
        time, flux,
        method="biweight",
        window_length=window_length,
        break_tolerance=break_tolerance,
        cval=cval,
        return_trend=True,
    )
    return detrended, trend


def check_exclusion(tmag: float, n_valid: int,
                    max_tmag: float = 6.0, min_cadences: int = 500
                    ) -> str | None:
    """Check if star should be excluded (PREP-06).
    
    Returns exclusion reason string or None if star passes.
    """
    if tmag < max_tmag:
        return "tmag<6"
    if n_valid < min_cadences:
        return "cadences<500"
    return None


def preprocess_single_star(tic_id: int, sector: int, config: dict) -> dict:
    """Preprocess a single star's light curve. Returns status dict."""
    paths = config["paths"]
    prep_cfg = config["preprocessing"]
    
    input_path = Path(paths["raw"]) / f"sector{sector}" / f"TIC_{tic_id}_raw.npz"
    output_path = Path(paths["preprocessed"]) / f"sector{sector}" / f"TIC_{tic_id}_preprocessed.npz"
    
    # Checkpoint: skip if already processed
    if output_path.exists():
        return {"tic_id": tic_id, "status": "skipped", "preprocess_s": 0.0}
    
    try:
        data = np.load(str(input_path))
        time = data["time"]
        flux = data["flux"]
        flux_err = data["flux_err"]
        quality = data["quality"]
        
        n_raw = len(time)
        
        # Step 1: Quality masking (PREP-01)
        time, flux, flux_err = apply_quality_mask(time, flux, flux_err, quality)
        
        # Step 2: Sigma-clipping (PREP-02)
        time, flux, flux_err = sigma_clip(
            flux, flux_err, time, sigma=prep_cfg["sigma_clip_threshold"]
        )
        
        n_valid = len(time)
        
        # Step 3: Check exclusion criteria (PREP-06)
        # tmag checked via catalogue; cadence count checked here
        if n_valid < prep_cfg["min_cadences"]:
            return {
                "tic_id": tic_id, "status": "excluded",
                "exclude_reason": "cadences<500",
                "n_cadences_raw": n_raw, "n_cadences_valid": n_valid,
                "preprocess_s": 0.0,
            }
        
        # Step 4: Normalize to median=1.0 (PREP-02)
        flux, flux_err = normalize_flux(flux, flux_err)
        
        # Step 5: Mask gap edges (PREP-05)
        time, flux, flux_err = mask_gap_edges(
            time, flux, flux_err,
            gap_threshold=prep_cfg["gap_threshold"]
        )
        
        # Step 6: Biweight detrending (PREP-03)
        detrended, trend = biweight_detrend(
            time, flux,
            window_length=prep_cfg["detrend_window_length"],
            break_tolerance=prep_cfg["detrend_break_tolerance"],
            cval=prep_cfg["detrend_cval"],
        )
        
        # Save preprocessed result
        np.savez_compressed(
            str(output_path),
            time=time.astype(np.float64),
            flux=detrended.astype(np.float32),
            flux_err=flux_err.astype(np.float32),
            trend=trend.astype(np.float32),
            flux_normalized=flux.astype(np.float32),
        )
        
        return {
            "tic_id": tic_id, "status": "success",
            "n_cadences_raw": n_raw, "n_cadences_valid": n_valid,
            "preprocess_s": 0.0,
        }
        
    except Exception as e:
        return {"tic_id": tic_id, "status": "failed", "error": str(e), "preprocess_s": 0.0}


def run_preprocess(config: dict, logger: logging.Logger) -> None:
    """Run preprocessing on all sectors with multiprocessing."""
    sectors = config["pipeline"]["sectors"]
    workers = config["pipeline"]["workers"]
    paths = config["paths"]

    for sector in sectors:
        raw_dir = Path(paths["raw"]) / f"sector{sector}"
        npz_files = list(raw_dir.glob("TIC_*_raw.npz"))
        tic_ids = [int(f.stem.split("_")[1]) for f in npz_files]
        
        logger.info(f"Preprocessing sector {sector}: {len(tic_ids)} stars")
        
        process_fn = partial(preprocess_single_star, sector=sector, config=config)
        
        success, failed, excluded = 0, 0, 0
        with Pool(processes=workers) as pool:
            results = list(tqdm(
                pool.imap(process_fn, tic_ids),
                total=len(tic_ids),
                desc=f"Preprocess S{sector}",
            ))
        
        for r in results:
            if r["status"] == "success":
                success += 1
            elif r["status"] == "failed":
                failed += 1
                logger.warning(
                    f"Preprocess failed TIC {r['tic_id']}: {r.get('error', 'unknown')}",
                    extra={"tic_id": r["tic_id"], "sector": sector, "status": "failed"}
                )
            elif r["status"] == "excluded":
                excluded += 1
        
        logger.info(
            f"Sector {sector} preprocessing: {success} success, "
            f"{excluded} excluded, {failed} failed"
        )
```
