## Threats
- **T1: TLS computation time exceeds available prep window** (severity: medium) — 60k stars at ~20s each = 333 hours single-threaded. Mitigation: Use stellar density priors (3-10× speedup), duration_grid_step=1.05, multiprocessing with 4 workers; checkpoint per star for Colab resilience.
- **T2: Harmonic confusion in multi-planet search** (severity: low) — EBs at period P produce TLS peaks at P/2, P/3, causing false multi-planet detections. Mitigation: BLS validation cross-checks period; Phase 2 classifier handles remaining confusion.
- **T3: Memory exhaustion with parallel TLS** (severity: low) — TLS working memory ~200MB per instance. Mitigation: Limit to 4 parallel workers (800MB total); well within 12GB Colab RAM.
</threat_model>

```python
"""Detection module — TLS period search, multi-planet iteration, BLS validation."""
import numpy as np
import pandas as pd
import logging
from pathlib import Path
from multiprocessing import Pool
from functools import partial
from tqdm import tqdm
from transitleastsquares import transitleastsquares, transit_mask
from astropy.timeseries import BoxLeastSquares
import astropy.units as u

from src.logging_utils import Timer


def compute_cdpp(flux: np.ndarray, cadence_minutes: float = 2.0,
                 window_hours: float = 1.0) -> float:
    """Compute 1-hour Combined Differential Photometric Precision in ppm (DET-04)."""
    points_per_window = int(window_hours * 60 / cadence_minutes)
    if len(flux) < points_per_window:
        return np.nan
    # Bin flux into window-sized chunks and compute scatter
    n_bins = len(flux) // points_per_window
    binned = np.array([
        np.nanmean(flux[i * points_per_window:(i + 1) * points_per_window])
        for i in range(n_bins)
    ])
    cdpp = np.nanstd(binned) * 1e6  # Convert to ppm
    return float(cdpp)


def run_tls_single(time: np.ndarray, flux: np.ndarray,
                   stellar_params: dict, config_det: dict) -> dict:
    """Run TLS on a single light curve. Returns results dict."""
    model = transitleastsquares(time, flux)
    
    # Build TLS kwargs with stellar priors if available
    tls_kwargs = {
        "period_min": config_det["period_min"],
        "period_max": config_det["period_max"],
        "n_transits_min": config_det["n_transits_min"],
        "oversampling_factor": config_det["oversampling_factor"],
        "duration_grid_step": config_det["duration_grid_step"],
        "use_threads": 1,
    }
    
    # Add stellar priors if available
    if stellar_params.get("radius") and stellar_params["radius"] > 0:
        tls_kwargs["R_star"] = stellar_params["radius"]
        tls_kwargs["R_star_min"] = stellar_params["radius"] * 0.9
        tls_kwargs["R_star_max"] = stellar_params["radius"] * 1.1
    if stellar_params.get("mass") and stellar_params["mass"] > 0:
        tls_kwargs["M_star"] = stellar_params["mass"]
        tls_kwargs["M_star_min"] = stellar_params["mass"] * 0.9
        tls_kwargs["M_star_max"] = stellar_params["mass"] * 1.1
    if stellar_params.get("ld_u1") is not None:
        tls_kwargs["u"] = [stellar_params["ld_u1"], stellar_params["ld_u2"]]
    
    results = model.power(**tls_kwargs)
    
    return {
        "period": float(results.period),
        "t0": float(results.T0),
        "duration": float(results.duration),
        "depth": float(results.depth) if results.depth else 0.0,
        "depth_ppm": float(results.depth) * 1e6 if results.depth else 0.0,
        "rp_rs": float(results.rp_rs) if hasattr(results, "rp_rs") else 0.0,
        "sde": float(results.SDE),
        "snr": float(results.snr) if hasattr(results, "snr") else 0.0,
        "snr_pink": float(results.snr_pink_per_transit) if hasattr(results, "snr_pink_per_transit") else 0.0,
        "n_transits": int(results.distinct_transit_count) if hasattr(results, "distinct_transit_count") else 0,
        "odd_even_mismatch": float(results.odd_even_mismatch) if hasattr(results, "odd_even_mismatch") else 0.0,
        "in_transit_mask": results.in_transit if hasattr(results, "in_transit") else None,
    }


def assign_sde_tier(sde: float, sde_discard: float = 5.0,
                    sde_subthreshold: float = 7.0) -> str:
    """Apply 3-tier SDE gating (DET-05)."""
    if sde >= sde_subthreshold:
        return "full"
    elif sde >= sde_discard:
        return "subthreshold"
    else:
        return "discard"


def validate_with_bls(time: np.ndarray, flux: np.ndarray,
                      flux_err: np.ndarray, tls_period: float,
                      tls_duration: float,
                      bls_snr_threshold: float = 6.0,
                      period_tolerance: float = 0.01) -> tuple:
    """BLS validation of TLS detection (DET-02). Returns (confirmed, bls_snr, bls_period)."""
    bls = BoxLeastSquares(time * u.day, flux, dy=flux_err)
    
    periods = np.linspace(tls_period * 0.95, tls_period * 1.05, 500) * u.day
    durations = np.array([tls_duration * 0.5, tls_duration, tls_duration * 1.5]) * u.day
    
    periodogram = bls.power(periods, durations, objective="snr")
    best_idx = np.argmax(periodogram.power)
    bls_snr = float(periodogram.power[best_idx])
    bls_period = float(periodogram.period[best_idx].value)
    
    period_match = abs(bls_period - tls_period) / tls_period < period_tolerance
    snr_ok = bls_snr > bls_snr_threshold
    
    return (period_match and snr_ok), bls_snr, bls_period


def detect_single_star(tic_id: int, sector: int, config: dict) -> list:
    """Run full detection on one star: iterative TLS + BLS validation.
    
    Returns list of detection dicts (one per planet found).
    """
    paths = config["paths"]
    det_cfg = config["detection"]
    
    input_path = Path(paths["preprocessed"]) / f"sector{sector}" / f"TIC_{tic_id}_preprocessed.npz"
    output_path = Path(paths["tls"]) / f"sector{sector}" / f"TIC_{tic_id}_tls.npz"
    
    # Checkpoint
    if output_path.exists():
        return [{"tic_id": tic_id, "status": "skipped"}]
    
    if not input_path.exists():
        return [{"tic_id": tic_id, "status": "no_input"}]
    
    try:
        data = np.load(str(input_path))
        time = data["time"]
        flux = data["flux"]
        flux_err = data["flux_err"]
        
        # Load stellar params from catalogue
        stellar_params = _load_stellar_params(tic_id, sector, config)
        
        # Compute CDPP (DET-04)
        cdpp = compute_cdpp(flux)
        
        # Iterative multi-planet search (DET-03)
        detections = []
        time_masked = time.copy()
        flux_masked = flux.copy()
        flux_err_masked = flux_err.copy()
        
        for iteration in range(det_cfg["max_iterations"]):
            if len(time_masked) < 100:
                break
            
            tls_result = run_tls_single(time_masked, flux_masked, stellar_params, det_cfg)
            
            if tls_result["sde"] < det_cfg["sde_discard"]:
                break  # No more signals above threshold
            
            # Assign SDE tier (DET-05)
            sde_tier = assign_sde_tier(
                tls_result["sde"],
                det_cfg["sde_discard"],
                det_cfg["sde_subthreshold"],
            )
            
            # BLS validation for full-pipeline candidates (DET-02)
            bls_confirmed, bls_snr, bls_period = False, 0.0, 0.0
            if sde_tier == "full":
                bls_confirmed, bls_snr, bls_period = validate_with_bls(
                    time_masked, flux_masked, flux_err_masked,
                    tls_result["period"], tls_result["duration"],
                    bls_snr_threshold=det_cfg["bls_snr_threshold"],
                    period_tolerance=det_cfg["bls_period_tolerance"],
                )
            
            detection = {
                "tic_id": tic_id,
                "sector": sector,
                "iteration": iteration + 1,
                "planet_index": iteration,
                "status": "success",
                **tls_result,
                "cdpp": cdpp,
                "sde_tier": sde_tier,
                "bls_confirmed": bls_confirmed,
                "bls_snr": bls_snr,
                "bls_period": bls_period,
            }
            # Remove in_transit_mask from saved dict (not serializable to parquet)
            detection.pop("in_transit_mask", None)
            detections.append(detection)
            
            # Mask found transit for next iteration (DET-03)
            intransit = transit_mask(
                time_masked, tls_result["period"],
                tls_result["duration"], tls_result["t0"]
            )
            time_masked = time_masked[~intransit]
            flux_masked = flux_masked[~intransit]
            flux_err_masked = flux_err_masked[~intransit]
        
        # Save TLS results to .npz
        if detections:
            save_dict = {}
            for i, det in enumerate(detections):
                for key, val in det.items():
                    if isinstance(val, (int, float, bool, str)):
                        save_dict[f"planet{i}_{key}"] = val
            np.savez_compressed(str(output_path), **save_dict)
        
        return detections if detections else [{"tic_id": tic_id, "status": "no_detection"}]
    
    except Exception as e:
        return [{"tic_id": tic_id, "status": "failed", "error": str(e)}]


def _load_stellar_params(tic_id: int, sector: int, config: dict) -> dict:
    """Load stellar parameters from catalogue Parquet for TLS priors."""
    cat_path = Path(config["paths"]["catalogue"]) / f"sector_{sector}.parquet"
    if not cat_path.exists():
        return {}
    try:
        df = pd.read_parquet(cat_path)
        row = df[df["tic_id"] == tic_id]
        if row.empty:
            return {}
        row = row.iloc[0]
        return {
            "radius": float(row.get("radius", 1.0)),
            "mass": float(row.get("mass", 1.0)),
            "ld_u1": float(row.get("ld_u1", 0.4)),
            "ld_u2": float(row.get("ld_u2", 0.26)),
        }
    except Exception:
        return {}


def run_detect(config: dict, logger: logging.Logger) -> None:
    """Run TLS detection on all preprocessed stars with multiprocessing."""
    sectors = config["pipeline"]["sectors"]
    workers = config["pipeline"]["workers"]
    paths = config["paths"]
    
    all_detections = []
    
    for sector in sectors:
        prep_dir = Path(paths["preprocessed"]) / f"sector{sector}"
        npz_files = list(prep_dir.glob("TIC_*_preprocessed.npz"))
        tic_ids = [int(f.stem.split("_")[1]) for f in npz_files]
        
        logger.info(f"Running TLS detection on sector {sector}: {len(tic_ids)} stars")
        
        process_fn = partial(detect_single_star, sector=sector, config=config)
        
        with Pool(processes=workers) as pool:
            results_nested = list(tqdm(
                pool.imap(process_fn, tic_ids),
                total=len(tic_ids),
                desc=f"TLS S{sector}",
            ))
        
        # Flatten results
        sector_detections = []
        success, failed, no_det = 0, 0, 0
        for result_list in results_nested:
            for r in result_list:
                if r.get("status") == "success":
                    sector_detections.append(r)
                    success += 1
                elif r.get("status") == "failed":
                    failed += 1
                    logger.warning(
                        f"TLS failed TIC {r['tic_id']}: {r.get('error', '')}",
                        extra={"tic_id": r["tic_id"], "sector": sector, "status": "failed"}
                    )
                elif r.get("status") == "no_detection":
                    no_det += 1
        
        all_detections.extend(sector_detections)
        
        # Summary
        n_full = sum(1 for d in sector_detections if d.get("sde_tier") == "full")
        n_sub = sum(1 for d in sector_detections if d.get("sde_tier") == "subthreshold")
        logger.info(
            f"Sector {sector} TLS: {success} detections ({n_full} full, {n_sub} sub-threshold), "
            f"{no_det} no signal, {failed} failed"
        )
    
    # Write TLS results Parquet
    if all_detections:
        tls_df = pd.DataFrame(all_detections)
        tls_path = Path(paths["catalogue"]) / "tls_results.parquet"
        tls_df.to_parquet(str(tls_path), index=False)
        logger.info(f"Saved {len(tls_df)} detections to {tls_path}")
        
        # Summary stats
        n_full_total = len(tls_df[tls_df["sde_tier"] == "full"])
        n_sub_total = len(tls_df[tls_df["sde_tier"] == "subthreshold"])
        logger.info(f"Total: {n_full_total} full-pipeline, {n_sub_total} sub-threshold candidates")
```
