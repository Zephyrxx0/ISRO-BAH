<objective>
Build the data ingestion module that bulk-downloads TESS 2-min cadence light curves from MAST for Sectors 1–3, stores them as .npz per TIC ID, queries TICv8 for stellar parameters and limb darkening, and produces a master Parquet catalogue.
</objective>

<threat_model>
## Threats
- **T1: MAST rate limiting causes incomplete download** (severity: medium) — Bulk queries may be throttled. Mitigation: Exponential backoff with jitter (base=2s, max 3 retries); pre-download during prep week; checkpoint per-star so partial runs resume.
- **T2: Corrupted FITS files from interrupted downloads** (severity: low) — Network interruption mid-download produces invalid files. Mitigation: Validate .npz after write by re-loading and checking array shapes; delete corrupted files on detection.
- **T3: TICv8 missing stellar parameters** (severity: low) — ~5-10% of targets lack Teff/logg. Mitigation: Use solar defaults (Teff=5778, logg=4.44, [M/H]=0.0) for missing values; flag in catalogue.
- **T4: Disk space exhaustion during bulk download** (severity: medium) — ~18GB of .npz files. Mitigation: Check available disk space before starting; log cumulative size; abort with clear message if <2GB remaining.
</threat_model>

```python
"""Data ingestion module — downloads TESS light curves from MAST."""
import numpy as np
import time
import random
import logging
from pathlib import Path
from tqdm import tqdm
from astroquery.mast import Observations, Catalogs
import lightkurve as lk

from src.logging_utils import Timer


def get_sector_targets(sector: int) -> list:
    """Query MAST for all 2-min cadence TIC IDs in a sector."""
    obs = Observations.query_criteria(
        obs_collection="TESS",
        dataproduct_type="timeseries",
        sequence_number=sector,
        t_exptime=[100, 130],
    )
    tic_ids = list(set(obs['target_name']))
    # Clean TIC IDs — remove 'TIC ' prefix if present, convert to int
    cleaned = []
    for t in tic_ids:
        t_str = str(t).replace("TIC ", "").strip()
        if t_str.isdigit():
            cleaned.append(int(t_str))
    return sorted(cleaned)


def download_lightcurve(tic_id: int, sector: int, output_dir: str,
                        max_retries: int = 3, backoff_base: float = 2.0,
                        backoff_jitter: float = 0.5) -> bool:
    """Download a single light curve with retry logic. Returns True on success."""
    output_path = Path(output_dir) / f"TIC_{tic_id}_raw.npz"
    if output_path.exists():
        return True  # Checkpoint: already downloaded

    for attempt in range(max_retries):
        try:
            sr = lk.search_lightcurve(
                f"TIC {tic_id}", mission="TESS",
                sector=sector, author="SPOC"
            )
            if len(sr) == 0:
                return False
            lc = sr[0].download(quality_bitmask="hard", flux_column="pdcsap_flux")
            np.savez_compressed(
                str(output_path),
                time=lc.time.value.astype(np.float64),
                flux=lc.flux.value.astype(np.float32),
                flux_err=lc.flux_err.value.astype(np.float32),
                quality=lc.quality.value.astype(np.int32),
                cadenceno=lc.cadenceno.value.astype(np.int32),
            )
            return True
        except Exception:
            if attempt < max_retries - 1:
                sleep_time = (backoff_base ** (attempt + 1)) + random.uniform(0, backoff_jitter)
                time.sleep(sleep_time)
            continue
    return False


def run_ingest(config: dict, logger: logging.Logger) -> None:
    """Run full ingestion for all configured sectors."""
    sectors = config["pipeline"]["sectors"]
    dl_cfg = config["download"]
    paths = config["paths"]

    for sector in sectors:
        output_dir = str(Path(paths["raw"]) / f"sector{sector}")
        logger.info(f"Ingesting sector {sector}", extra={"sector": sector})
        tic_ids = get_sector_targets(sector)
        logger.info(f"Found {len(tic_ids)} targets in sector {sector}")

        success, failed = 0, 0
        for tic_id in tqdm(tic_ids, desc=f"Sector {sector}"):
            with Timer() as t:
                ok = download_lightcurve(
                    tic_id, sector, output_dir,
                    max_retries=dl_cfg["max_retries"],
                    backoff_base=dl_cfg["backoff_base"],
                    backoff_jitter=dl_cfg["backoff_jitter"],
                )
            if ok:
                success += 1
                logger.info(
                    f"Downloaded TIC {tic_id}",
                    extra={"tic_id": tic_id, "sector": sector,
                           "duration_s": round(t.elapsed, 2), "status": "success"}
                )
            else:
                failed += 1
                logger.warning(
                    f"Failed TIC {tic_id}",
                    extra={"tic_id": tic_id, "sector": sector, "status": "failed"}
                )
            time.sleep(dl_cfg["rate_limit_sleep"])

        logger.info(f"Sector {sector} complete: {success} success, {failed} failed")
```

"""Catalogue management — TICv8 queries and Parquet I/O."""
import numpy as np
import pandas as pd
import logging
from pathlib import Path
from astroquery.mast import Catalogs
from tqdm import tqdm


MASTER_COLUMNS = [
    "tic_id", "sector", "ra", "dec", "tmag", "teff", "logg",
    "radius", "mass", "metallicity", "crowdsap",
    "ld_u1", "ld_u2",
    "n_cadences_raw", "n_cadences_valid", "cdpp_1hr",
    "preprocess_status", "exclude_reason",
    "download_s", "preprocess_s", "tls_s",
]


def query_tic_batch(tic_ids: list, batch_size: int = 500) -> pd.DataFrame:
    """Query TICv8 stellar parameters in batches."""
    all_rows = []
    for i in tqdm(range(0, len(tic_ids), batch_size), desc="TICv8 query"):
        batch = tic_ids[i:i + batch_size]
        id_str = ",".join(str(t) for t in batch)
        try:
            result = Catalogs.query_criteria(catalog="TIC", ID=id_str)
            df = result.to_pandas()
            all_rows.append(df)
        except Exception:
            continue
    if not all_rows:
        return pd.DataFrame()
    return pd.concat(all_rows, ignore_index=True)


def build_sector_catalogue(tic_ids: list, sector: int,
                           ld_table: pd.DataFrame,
                           output_path: str) -> pd.DataFrame:
    """Build per-sector Parquet catalogue from TICv8 data."""
    raw = query_tic_batch(tic_ids)
    if raw.empty:
        return pd.DataFrame(columns=MASTER_COLUMNS)

    from src.limb_darkening import interpolate_ld

    rows = []
    for _, star in raw.iterrows():
        tic_id = int(star.get("ID", 0))
        teff = float(star.get("Teff", 5778))
        logg_val = float(star.get("logg", 4.44))
        met = float(star.get("MH", 0.0)) if pd.notna(star.get("MH")) else 0.0
        u1, u2 = interpolate_ld(teff, logg_val, met, ld_table)
        rows.append({
            "tic_id": tic_id,
            "sector": sector,
            "ra": float(star.get("ra", 0.0)),
            "dec": float(star.get("dec", 0.0)),
            "tmag": float(star.get("Tmag", 99.0)),
            "teff": teff,
            "logg": logg_val,
            "radius": float(star.get("rad", 1.0)),
            "mass": float(star.get("mass", 1.0)),
            "metallicity": met,
            "crowdsap": float(star.get("contratio", 0.0)),
            "ld_u1": u1,
            "ld_u2": u2,
            "n_cadences_raw": 0,
            "n_cadences_valid": 0,
            "cdpp_1hr": 0.0,
            "preprocess_status": "pending",
            "exclude_reason": None,
            "download_s": 0.0,
            "preprocess_s": 0.0,
            "tls_s": 0.0,
        })

    df = pd.DataFrame(rows)
    df.to_parquet(output_path, index=False)
    return df


def merge_master_catalogue(catalogue_dir: str, sectors: list) -> None:
    """Merge per-sector Parquet files into master.parquet."""
    dfs = []
    for sector in sectors:
        path = Path(catalogue_dir) / f"sector_{sector}.parquet"
        if path.exists():
            dfs.append(pd.read_parquet(path))
    if dfs:
        master = pd.concat(dfs, ignore_index=True)
        master.to_parquet(Path(catalogue_dir) / "master.parquet", index=False)
```

"""Limb darkening coefficient interpolation from Claret (2017) tables."""
import numpy as np
import pandas as pd
from pathlib import Path
from scipy.interpolate import LinearNDInterpolator


# Solar defaults for missing parameters
SOLAR_TEFF = 5778.0
SOLAR_LOGG = 4.44
SOLAR_MET = 0.0
DEFAULT_LD = [0.4, 0.26]  # Fallback for interpolation failures


def load_ld_table(path: str = "data/reference/claret2017_tess_ld.csv") -> pd.DataFrame:
    """Load pre-downloaded Claret (2017) TESS limb darkening table."""
    return pd.read_csv(path)


def build_ld_interpolator(ld_table: pd.DataFrame):
    """Build LinearNDInterpolator from Claret table."""
    points = np.column_stack([
        ld_table["Teff"].values,
        ld_table["logg"].values,
        ld_table["Z"].values,
    ])
    u1_interp = LinearNDInterpolator(points, ld_table["u1"].values)
    u2_interp = LinearNDInterpolator(points, ld_table["u2"].values)
    return u1_interp, u2_interp


_interpolators = None


def interpolate_ld(teff: float, logg: float, met: float,
                   ld_table: pd.DataFrame) -> tuple:
    """Interpolate quadratic LD coefficients [u1, u2] for TESS bandpass.
    
    Returns solar defaults if parameters are NaN or outside table bounds.
    """
    global _interpolators
    if _interpolators is None:
        _interpolators = build_ld_interpolator(ld_table)
    u1_interp, u2_interp = _interpolators

    if np.isnan(teff) or np.isnan(logg):
        return DEFAULT_LD[0], DEFAULT_LD[1]

    u1 = float(u1_interp(teff, logg, met))
    u2 = float(u2_interp(teff, logg, met))

    if np.isnan(u1) or np.isnan(u2):
        return DEFAULT_LD[0], DEFAULT_LD[1]

    return u1, u2

"""Download Target Pixel Files (TPFs) via TESScut for centroid analysis on SDE≥7 candidates.

Usage: python scripts/download_tpf.py --candidates data/tls/candidates_sde7.parquet --sector 1
Run AFTER TLS detection (Wave 2) identifies SDE≥7 candidates.
"""
import argparse
import time
import random
import numpy as np
import pandas as pd
from pathlib import Path
import lightkurve as lk
from tqdm import tqdm


def download_tpf(tic_id: int, sector: int, output_dir: Path,
                 max_retries: int = 3, backoff_base: float = 2.0) -> bool:
    """Download TPF for a single target with retry logic."""
    output_path = output_dir / f"TIC_{tic_id}_tpf_s{sector}.npz"
    if output_path.exists():
        return True  # Checkpoint

    for attempt in range(max_retries):
        try:
            sr = lk.search_targetpixelfile(
                f"TIC {tic_id}", mission="TESS",
                sector=sector, author="SPOC"
            )
            if len(sr) == 0:
                return False
            tpf = sr[0].download()
            np.savez_compressed(
                str(output_path),
                flux=tpf.flux.value.astype(np.float32),
                flux_err=tpf.flux_err.value.astype(np.float32),
                time=tpf.time.value.astype(np.float64),
                column=int(tpf.column),
                row=int(tpf.row),
                wcs_header=str(tpf.wcs.to_header()),
            )
            return True
        except Exception:
            if attempt < max_retries - 1:
                sleep_time = (backoff_base ** (attempt + 1)) + random.uniform(0, 0.5)
                time.sleep(sleep_time)
            continue
    return False


def main():
    parser = argparse.ArgumentParser(description="Download TPFs for centroid analysis")
    parser.add_argument("--candidates", type=str, required=True,
                        help="Path to Parquet with SDE≥7 candidates")
    parser.add_argument("--sector", type=int, required=True)
    parser.add_argument("--config", type=str, default="config.yaml")
    args = parser.parse_args()

    candidates = pd.read_parquet(args.candidates)
    sde7 = candidates[candidates["sde"] >= 7.0]
    output_dir = Path(f"data/tpf/sector{args.sector}")
    output_dir.mkdir(parents=True, exist_ok=True)

    print(f"Downloading TPFs for {len(sde7)} SDE≥7 candidates in sector {args.sector}")
    success, failed = 0, 0
    for _, row in tqdm(sde7.iterrows(), total=len(sde7)):
        tic_id = int(row["tic_id"])
        if download_tpf(tic_id, args.sector, output_dir):
            success += 1
        else:
            failed += 1
    print(f"TPF download complete: {success} success, {failed} failed")


if __name__ == "__main__":
    main()
```

"""Download Kepler DR24 TCE catalog (AstroNet labeled dataset) for CNN pre-training.

Downloads 34,032 labeled TCE samples from the Kepler Q1-Q17 DR24 catalog.
Run during prep week: python scripts/download_kepler_dr24.py

DATA-04: Pipeline downloads Kepler DR24 TCE catalog (AstroNet dataset, 34,032 labeled samples) for CNN pre-training.
"""
import numpy as np
import pandas as pd
from pathlib import Path
from astroquery.mast import Observations
from tqdm import tqdm
import lightkurve as lk
import time
import random


DR24_CATALOG_URL = "https://exoplanetarchive.ipac.caltech.edu/cgi-bin/nstedAPI/nph-nstedAPI?table=q1_q17_dr24_tce&select=*&format=csv"


def download_dr24_catalog(output_dir: Path) -> pd.DataFrame:
    """Download the DR24 TCE table from NASA Exoplanet Archive."""
    catalog_path = output_dir / "kepler_dr24_tce.csv"
    if catalog_path.exists():
        print(f"Using cached catalog: {catalog_path}")
        return pd.read_csv(catalog_path)

    print("Downloading Kepler DR24 TCE catalog from NASA Exoplanet Archive...")
    import urllib.request
    urllib.request.urlretrieve(DR24_CATALOG_URL, str(catalog_path))
    df = pd.read_csv(catalog_path)
    print(f"Downloaded {len(df)} TCEs to {catalog_path}")
    return df


def download_kepler_lightcurves(catalog: pd.DataFrame, output_dir: Path,
                                 max_targets: int = 34032,
                                 max_retries: int = 3) -> None:
    """Download Kepler light curves for TCE targets."""
    lc_dir = output_dir / "lightcurves"
    lc_dir.mkdir(parents=True, exist_ok=True)

    kic_ids = catalog["kepid"].unique()[:max_targets]
    print(f"Downloading light curves for {len(kic_ids)} Kepler targets...")

    success, failed = 0, 0
    for kic_id in tqdm(kic_ids):
        output_path = lc_dir / f"KIC_{kic_id}.npz"
        if output_path.exists():
            success += 1
            continue

        for attempt in range(max_retries):
            try:
                sr = lk.search_lightcurve(
                    f"KIC {kic_id}", mission="Kepler", author="Kepler"
                )
                if len(sr) == 0:
                    break
                lc_collection = sr.download_all()
                lc = lc_collection.stitch()
                np.savez_compressed(
                    str(output_path),
                    time=lc.time.value.astype(np.float64),
                    flux=lc.flux.value.astype(np.float32),
                    flux_err=lc.flux_err.value.astype(np.float32),
                )
                success += 1
                break
            except Exception:
                if attempt < max_retries - 1:
                    time.sleep((2 ** (attempt + 1)) + random.uniform(0, 0.5))
                else:
                    failed += 1

    print(f"Kepler download complete: {success} success, {failed} failed")


def main():
    output_dir = Path("data/kepler_dr24")
    output_dir.mkdir(parents=True, exist_ok=True)

    catalog = download_dr24_catalog(output_dir)
    download_kepler_lightcurves(catalog, output_dir)


if __name__ == "__main__":
    main()
