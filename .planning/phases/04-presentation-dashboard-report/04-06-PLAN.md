2. Update run_pipeline.py to add Phase 4 step at the end:
   - After Phase 3 (characterization) completes, add:
   ```python
   # Phase 4: Generate presentation outputs
   from pathlib import Path
   from src.report import generate_catalogue, generate_report, generate_candidates_json, generate_star_jsons
   import shutil

   def run_phase4_presentation(outputs_dir: Path, dashboard_dir: Path):
       """Generate all Phase 4 deliverables."""
       print("Phase 4: Generating presentation outputs...")

       # 4a: Candidate catalogue CSV
       generate_catalogue(
           results_dir=outputs_dir,
           output_path=outputs_dir / 'catalogue.csv'
       )
       print(f"  ✓ Catalogue CSV: {outputs_dir / 'catalogue.csv'}")

       # 4b: Dashboard data (JSON)
       generate_candidates_json(
           results_dir=outputs_dir,
           output_path=outputs_dir / 'candidates.json'
       )
       stars_dir = outputs_dir / 'stars'
       stars_dir.mkdir(exist_ok=True)
       n_stars = generate_star_jsons(
           results_dir=outputs_dir,
           output_dir=stars_dir
       )
       print(f"  ✓ Dashboard data: candidates.json + {n_stars} star JSONs")

       # 4c: PDF report
       generate_report(
           outputs_dir=outputs_dir,
           output_path=outputs_dir / 'report.pdf'
       )
       print(f"  ✓ PDF report: {outputs_dir / 'report.pdf'}")

       # 4d: Copy plots to dashboard public/ for static build
       plots_dest = dashboard_dir / 'public' / 'plots'
       if plots_dest.exists():
           shutil.rmtree(plots_dest)
       if (outputs_dir / 'plots').exists():
           shutil.copytree(outputs_dir / 'plots', plots_dest)
           print(f"  ✓ Plots copied to {plots_dest}")

       # 4e: Copy data to dashboard outputs/ for SSG build
       dash_outputs = dashboard_dir / 'outputs'
       dash_outputs.mkdir(exist_ok=True)
       shutil.copy2(outputs_dir / 'candidates.json', dash_outputs)
       stars_dash = dash_outputs / 'stars'
       if stars_dash.exists():
           shutil.rmtree(stars_dash)
       shutil.copytree(stars_dir, stars_dash)
       print(f"  ✓ Data copied to dashboard/outputs/")

       print("Phase 4: Presentation outputs complete.")
   ```

Create Makefile at project root:

```makefile
.PHONY: all pipeline dashboard serve clean

# Default: run entire pipeline and build dashboard
all: pipeline dashboard

# Run Python pipeline (Phases 1-4 data generation)
pipeline:
	python run_pipeline.py --sectors 1,2,3

# Build Next.js static dashboard (requires pipeline outputs)
dashboard: 
	cd dashboard && npx next build

# Serve the built dashboard locally
serve:
	cd dashboard && npx serve out -l 3000

# Clean all generated outputs
clean:
	rm -rf outputs/
	rm -rf dashboard/out/
	rm -rf dashboard/public/plots/
	rm -rf dashboard/outputs/

# Development: run dashboard in dev mode
dev:
	cd dashboard && npx next dev

# Quick: skip pipeline, just rebuild dashboard (assumes outputs/ exists)
dashboard-only:
	cd dashboard && npx next build
```

Key design decisions:
- `pipeline` target runs run_pipeline.py which handles ALL Phase 4 data generation + file copying
- `dashboard` target ONLY runs next build (data already in place from pipeline step)
- `serve` uses npx serve for static file serving (no dev server needed for judges)
- `clean` removes all generated artifacts
- `dashboard-only` for iterating on frontend without re-running Python pipeline
